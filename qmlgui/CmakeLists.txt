# qmlgui/CMakeLists.txt

project(study_assistant_qml LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS
    Core
    Quick
    Gui
    Qml
    Network
)

# ---- QML GUI executable ----
qt_add_executable(study_assistant_qml
    main.cpp
    BoardManager.cpp
    BoardManager.h
    FlashcardModel.cpp
    FlashcardModel.h
    StudyController.cpp
    StudyController.h
)

target_link_libraries(study_assistant_qml
    PRIVATE
        Qt6::Core
        Qt6::Quick
        Qt6::Gui
        Qt6::Qml
        Qt6::Network
        core         # <-- the backend library from the top-level CMakeLists
)

qt_add_qml_module(study_assistant_qml
    URI StudyAssistant
    VERSION 1.0
    QML_FILES
        qml/main.qml
        qml/MyBoardsPage.qml
        qml/BoardPage.qml
        qml/StudyFlashcardsPage.qml
        qml/AskAiPage.qml
)
